@startuml
allowmixing

package "Gestión de Menú - Categorías" {

  class Categoria <<Entity>> {
    + id : smallint
    + nombre : nvarchar(20)
    + ordenEnPantalla : tinyint
    + color : int
    + estaInactivo : bit
    + parentId : smallint
  }

  interface FormularioCategoria <<Boundary>> {
    + mostrarFormulario()
    + mostrarMensaje(msg: string)
    + solicitarConfirmacion() : bool
  }

  class GestorCategoria <<Control>> {
    + agregarCategoria(datos: Categoria)
    + modificarCategoria(datos: Categoria)
    + eliminarCategoria(id: smallint)
  }

}

actor Admin

Admin --> FormularioCategoria: Interactúa
FormularioCategoria --> GestorCategoria: Envía peticiones
GestorCategoria --> Categoria: Realiza operaciones

@enduml
